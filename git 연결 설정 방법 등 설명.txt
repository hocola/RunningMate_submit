git 연결 설정 방법 등 설명

----------------------------------------------------------------------

1단계: Git 초기 설정 및 원격 저장소 연결
bash
# 1. 해당 폴더로 이동
cd C:\Users\user\Desktop\running_app

# 2. Git 초기화 (이미 되어있다면 건너뛰기)
git init

# 3. 원격 저장소 연결
git remote add origin https://github.com/hocola/RunningMate.git

# 4. 원격 저장소 확인
git remote -v


2단계: Main 브랜치 가져오기
bash
# 1. 원격 저장소의 모든 브랜치 정보 가져오기
git fetch origin

# 2. Main 브랜치로 체크아웃
git checkout main

# 3. 원격의 main 브랜치와 동기화
git pull origin main

3단계: 새 브랜치 3개 생성
bash
# 1. jeong 브랜치 생성 및 이동
git checkout -b jeong // 이 명령어는 현재 브랜치(예: main)의 모든 내용을 복사

# 2. song 브랜치 생성
git checkout main
git checkout -b song

# 3. kim 브랜치 생성
git checkout main
git checkout -b kim

# 4. 다시 jeong 브랜치로 이동 (작업용)
git checkout jeong

4단계: 브랜치를 원격에 푸시
bash
# 각 브랜치를 원격에 올리기
git push origin jeong
git push origin song
git push origin kim

5단계: 일상적인 작업 플로우
jeong 브랜치에서 작업할 때:
bash
# 1. jeong 브랜치에 있는지 확인
git branch  # 현재 브랜치 확인 (* 표시가 있는 것)

# 2. 파일 수정 후 변경사항 확인
git status

# 3. 변경된 파일 추가
git add .  # 모든 파일
# 또는
git add 파일명  # 특정 파일만

# 4. 커밋 (변경사항 저장)
git commit -m "상단바 스타일 수정(시간/배터리 잘 보이게 변경)"

# 5. 원격 저장소에 업로드
git push origin jeong

작업 완료 후 main에 합치기:
bash
# 1. main 브랜치로 이동
git checkout main

# 2. main을 최신 상태로 업데이트
git pull origin main

# 3. jeong 브랜치의 내용을 main에 병합
git merge jeong

# 4. main을 원격에 푸시
git push origin main

유용한 Git 명령어
bash
# 현재 브랜치 확인
git branch

# 모든 브랜치 확인 (원격 포함)
git branch -a

# 브랜치 전환
git checkout 브랜치명

# 변경사항 확인
git status

# 커밋 기록 보기
git log

# 원격 저장소 최신 정보 가져오기 (병합 안함)
git fetch origin
주의사항

항상 작업 전에: git pull origin jeong 으로 최신 상태 유지
커밋 메시지: 명확하게 작성 (예: "로그인 기능 추가", "버그 수정: 맵 로딩 오류")
자주 커밋: 작은 단위로 자주 커밋하는 것이 좋습니다
main 브랜치: 최종 완성본만 올리기
